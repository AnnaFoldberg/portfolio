{%- comment -%} Lang Switcher (jekyll-polyglot) - Determines current_lang from page.lang, else URL segment, else default - Resolves counterpart by ref/slug within pages or current collection - Default language uses "/" as home; others use "//" {%- endcomment -%} {%- assign langs = site.languages -%} {%- comment -%} Robust current language (do NOT use site.active_lang) {%- endcomment -%} {%- assign current_lang = page.lang -%} {%- if current_lang == nil or current_lang == '' -%} {%- assign seg = page.url | split:'/' | slice: 1, 1 | first -%} {%- if seg and seg != '' and langs contains seg -%} {%- assign current_lang = seg -%} {%- else -%} {%- assign current_lang = site.default_lang -%} {%- endif -%} {%- endif -%}
    {%- for L in langs -%} {%- comment -%} Display label (from _data/locales/.yml if present) {%- endcomment -%} {%- assign label = L | upcase -%} {%- if site.data.locales and site.data.locales[L] and site.data.locales[L].lang_name -%} {%- assign label = site.data.locales[L].lang_name -%} {%- endif -%} {%- assign target_url = nil -%} {%- comment -%} 1) Try matching a normal page by ref {%- endcomment -%} {%- if page.ref -%} {%- assign page_candidate = site.pages | where: "lang", L | where: "ref", page.ref | first -%} {%- if page_candidate and page_candidate.url -%} {%- assign target_url = page_candidate.url -%} {%- endif -%} {%- endif -%} {%- comment -%} 2) Try matching within the current collection (for collection items) {%- endcomment -%} {%- if target_url == nil and page.collection -%} {%- assign items = site[page.collection] | where: "lang", L -%} {%- assign candidate = nil -%} {%- if page.ref -%} {%- assign candidate = items | where: "ref", page.ref | first -%} {%- endif -%} {%- if candidate == nil -%} {%- assign pslug = page.slug | default: page.name | split: "." | first -%} {%- assign candidate = items | where: "slug", pslug | first -%} {%- endif -%} {%- if candidate and candidate.url -%} {%- assign target_url = candidate.url -%} {%- endif -%} {%- endif -%} {%- comment -%} 3) Fallback to language home (default -> "/") {%- endcomment -%} {%- if target_url == nil -%} {%- if L == site.default_lang -%} {%- assign target_url = "/" -%} {%- else -%} {%- assign target_url = "/" | append: L | append: "/" -%} {%- endif -%} {%- endif -%}
