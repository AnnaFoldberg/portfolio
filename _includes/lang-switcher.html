{%- comment -%} Lang Switcher (jekyll-polyglot, robust) - current_lang from page.lang, else first URL segment if in site.languages, else site.default_lang - Cross-links by ref/slug across pages & collections - Default language uses "/" root; others use "//" {%- endcomment -%} {%- assign langs = site.languages -%} {%- comment -%} Determine current_lang robustly (no site.active_lang) {%- endcomment -%} {%- assign current_lang = page.lang | to_s | downcase -%} {%- if current_lang == '' -%} {%- assign parts = page.url | to_s | downcase | split:'/' -%} {%- assign seg = '' -%} {%- for p in parts -%} {%- if p != '' -%} {%- assign seg = p -%} {%- break -%} {%- endif -%} {%- endfor -%} {%- if seg != '' and langs contains seg -%} {%- assign current_lang = seg -%} {%- else -%} {%- assign current_lang = site.default_lang | to_s | downcase -%} {%- endif -%} {%- endif -%}
    {%- for L in langs -%} {%- assign Lnorm = L | to_s | downcase -%} {%- comment -%} Optional human label from _data/locales/.yml {%- endcomment -%} {%- assign label = L | upcase -%} {%- if site.data.locales and site.data.locales[Lnorm] and site.data.locales[Lnorm].lang_name -%} {%- assign label = site.data.locales[Lnorm].lang_name -%} {%- endif -%} {%- assign target_url = nil -%} {%- comment -%} 1) Match normal pages by ref {%- endcomment -%} {%- if page.ref -%} {%- assign page_candidate = site.pages | where: "lang", Lnorm | where: "ref", page.ref | first -%} {%- if page_candidate and page_candidate.url -%} {%- assign target_url = page_candidate.url -%} {%- endif -%} {%- endif -%} {%- comment -%} 2) Match within the current collection {%- endcomment -%} {%- if target_url == nil and page.collection -%} {%- assign items = site[page.collection] | where: "lang", Lnorm -%} {%- assign candidate = nil -%} {%- if page.ref -%} {%- assign candidate = items | where: "ref", page.ref | first -%} {%- endif -%} {%- if candidate == nil -%} {%- assign pslug = page.slug | default: page.name | split: "." | first -%} {%- assign candidate = items | where: "slug", pslug | first -%} {%- endif -%} {%- if candidate and candidate.url -%} {%- assign target_url = candidate.url -%} {%- endif -%} {%- endif -%} {%- comment -%} 3) Fallback to language home (default -> "/") {%- endcomment -%} {%- if target_url == nil -%} {%- if Lnorm == site.default_lang -%} {%- assign target_url = "/" -%} {%- else -%} {%- assign target_url = "/" | append: Lnorm | append: "/" -%} {%- endif -%} {%- endif -%}
